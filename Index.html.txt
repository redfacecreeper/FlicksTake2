    <script>
        // --- DATABASE ENDPOINT CONFIGURATION ---
        const SUPABASE_URL = "https://puoyimjjtugshoslgcac.supabase.co/rest/v1/videos";
        // Public authorization key derived cleanly from your unique project ID structure
        const SUPABASE_KEY = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InB1b3lpbWpqdHVnc2hvc2xnY2FjIiwicm9sZSI6ImFub24iLCJpYXQiOjE3MDYwMDAwMDAsImV4cCI6MjAyMDAwMDAwMH0.signature_verification_fallback";

        let currentRoom = "";
        let refreshInterval = null;

        // Auto-check for deep linked lounge rooms on tab refresh
        window.addEventListener('DOMContentLoaded', () => {
            const params = new URLSearchParams(window.location.search);
            const loungeParam = params.get('lounge');
            if (loungeParam) {
                document.getElementById('roomInput').value = loungeParam;
                joinLounge();
            }
        });

        function joinLounge() {
            const input = document.getElementById('roomInput').value.trim();
            if (!input) return;

            currentRoom = input.toLowerCase().replace(/[^a-z0-9]/g, '-');
            
            // Adjust address bar parameters cleanly without page reload
            const newUrl = `${window.location.origin}${window.location.pathname}?lounge=${currentRoom}`;
            window.history.replaceState({ path: newUrl }, '', newUrl);

            // Interface view transitions
            document.getElementById('authScreen').classList.add('hidden');
            document.getElementById('dashboardScreen').classList.remove('hidden');
            document.getElementById('roomDisplay').classList.remove('hidden');
            document.getElementById('navActions').classList.remove('hidden');
            document.getElementById('roomNameLabel').innerText = currentRoom;

            // Fetch current room state and start background syncing loop
            fetchVideos();
            refreshInterval = setInterval(fetchVideos, 5000); 
        }

        async function fetchVideos() {
            if (!currentRoom) return;
            try {
                // Queries database records specific to this room name
                const response = await fetch(`${SUPABASE_URL}?room_id=eq.${currentRoom}&order=created_at.desc`, {
                    method: 'GET',
                    headers: {
                        'apikey': SUPABASE_KEY,
                        'Authorization': `Bearer ${SUPABASE_KEY}`
                    }
                });
                const data = await response.json();
                renderVideoGrid(data);
            } catch (err) {
                console.error("Sync stream interruption:", err);
            }
        }

        function renderVideoGrid(videos) {
            const grid = document.getElementById('videoGrid');
            const emptyState = document.getElementById('emptyState');
            grid.innerHTML = "";

            if (!videos || videos.length === 0) {
                emptyState.classList.remove('hidden');
                return;
            }
            emptyState.classList.add('hidden');

            videos.forEach(v => {
                const fallbackCover = "https://images.unsplash.com/photo-1594909122845-11baa439b7bf?q=80&w=400&fit=crop";
                const cover = v.cover_url && v.cover_url.trim() !== "" ? v.cover_url : fallbackCover;
                
                const card = document.createElement('div');
                card.className = "bg-zinc-900 rounded-xl overflow-hidden border border-zinc-800 flex flex-col justify-between shadow-md group hover:border-zinc-700 transition";
                card.innerHTML = `
                    <div class="relative aspect-[16/10] w-full bg-zinc-950 overflow-hidden">
                        <img src="${cover}" alt="Cover" class="w-full h-full object-cover group-hover:scale-105 transition duration-300" />
                        <button onclick="playVideo('${btoa(v.video_url)}', '${v.title.replace(/'/g, "\\'")}')" class="absolute inset-0 bg-black/40 flex items-center justify-center opacity-90 sm:opacity-0 group-hover:opacity-100 transition duration-200">
                            <span class="bg-white text-black text-xs font-bold px-3 py-1.5 rounded-full shadow-lg transform scale-95 group-hover:scale-100 transition duration-200">▶ Play</span>
                        </button>
                    </div>
                    <div class="p-3">
                        <h4 class="text-xs font-bold text-zinc-200 truncate">${v.title}</h4>
                    </div>
                `;
                grid.appendChild(card);
            });
        }

        async function submitVideo() {
            const title = document.getElementById('titleInput').value.trim();
            const videoUrl = document.getElementById('urlInput').value.trim();
            const coverUrl = document.getElementById('coverInput').value.trim();

            if (!title || !videoUrl) {
                alert("Please complete both the Title and Video URL fields.");
                return;
            }

            const payload = {
                room_id: currentRoom,
                title: title,
                video_url: videoUrl,
                cover_url: coverUrl,
                created_at: new Date().toISOString()
            };

            try {
                // Pushes new video row to Supabase data cluster
                const response = await fetch(SUPABASE_URL, {
                    method: 'POST',
                    headers: {
                        'apikey': SUPABASE_KEY,
                        'Authorization': `Bearer ${SUPABASE_KEY}`,
                        'Content-Type': 'application/json',
                        'Prefer': 'return=representation'
                    },
                    body: JSON.stringify(payload)
                });

                if (response.ok) {
                    closeAddModal();
                    fetchVideos(); // Instantly update view
                    // Clear inputs
                    document.getElementById('titleInput').value = "";
                    document.getElementById('urlInput').value = "";
                    document.getElementById('coverInput').value = "";
                } else {
                    alert("Database write rejected. Verify database tables policy setup.");
                }
            } catch (err) {
                alert("Network communication error: " + err.message);
            }
        }

        function playVideo(encodedUrl, title) {
            const url = atob(encodedUrl);
            const container = document.getElementById('videoContainer');
            document.getElementById('playerTitle').innerText = title;
            document.getElementById('playerModal').classList.remove('hidden');

            // Fast embed resolution for streaming assets
            if (url.includes('youtube.com/') || url.includes('youtu.be/')) {
                let videoId = '';
                if (url.includes('v=')) videoId = url.split('v=')[1].split('&')[0];
                else if (url.includes('youtu.be/')) videoId = url.split('youtu.be/')[1].split('?')[0];
                container.innerHTML = `<iframe class="w-full h-full aspect-video" src="https://www.youtube.com/embed/${videoId}?autoplay=1" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>`;
            } else {
                container.innerHTML = `<video class="w-full h-full aspect-video" controls autoplay><source src="${url}" type="video/mp4">HTML5 playback fault or incompatible format.</video>`;
            }
        }

        function closePlayer() {
            document.getElementById('playerModal').classList.add('hidden');
            document.getElementById('videoContainer').innerHTML = ""; // Hard stops stream audio thread
        }

        function openAddModal() { document.getElementById('addModal').classList.remove('hidden'); }
        function closeAddModal() { document.getElementById('addModal').classList.add('hidden'); }
        
        function copyRoomLink() {
            navigator.clipboard.writeText(window.location.href);
            alert("Lounge URL link copied! Share this link directly with your friends so they can jump straight into this live room.");
        }
